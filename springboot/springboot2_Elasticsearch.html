@Document(indexName = "SpringContact")
public class SpringContact {

	@Id
	private Long id;

	public Long getId() {
		return this.id;
	}

		this.id = id;
	}

		this.userName = userName;
	}

	private String mobile;

	public String getMobile() {
		return this.mobile;
	}

	public void setMobile(String mobile) {
		this.mobile = mobile;
	}

	private String tel;

	public String getTel() {
		return this.tel;
	}

	public void setTel(String tel) {

	}

	private String fax;

	public String getFax() {
		return this.fax;
	}

	public void setFax(String fax) {
		this.fax = fax;
	}
}

实现SpringContactService中定义的数据访问操作

@Service
public class SpringContactServiceImpl implements SpringContactService {

	@Autowired
	private SpringContactDao dao;
	
	@Override
	public void create(SpringContact entity) {
		this.dao.save(entity);
	}

	@Override
	public void update(SpringContact entity) {
		this.dao.save(entity);
	}

	@Override
	public void deleteByUserName(String userName) {
		SpringContact entity=this.dao.findByUserName(userName);
		this.dao.delete(entity);
	}

	@Override
	public SpringContact getByUserName(String userName) {
		return this.dao.findByUserName(userName);
	}

	@Override
	public Long getTotalNumbers() {
		return this.dao.count();
	}

}

创建对SpringContactService的单元测试用例

@RunWith(SpringRunner.class)
@SpringBootTest
class SpringContactServiceTest {

	@Autowired
	private SpringContactService springContactService;

	@Test
	void test() {

		for (int i = 1; i < 6; i++) {
			SpringContact entity = new SpringContact();
			entity.setId(new Long(i));
			entity.setUserName("SpringSongs"+i);
			entity.setMobile("12345678911");
			entity.setTel("0777-955888");
			entity.setFax("0777-955888");
			this.springContactService.create(entity);
		}

		Long counts = this.springContactService.getTotalNumbers();
		Assert.assertEquals(new Long(5), counts);
		
		
		SpringContact springContact = this.springContactService.getByUserName("SpringSongs1");
		Assert.assertEquals("SpringSongs1", springContact.getUserName());

		
		this.springContactService.deleteByUserName("SpringSongs1");
		counts = this.springContactService.getTotalNumbers();
		Assert.assertEquals(new Long(4), counts);
		
	}

}
